#include <stdio.h>
#include <stdlib.h> 
#include <time.h> 
int main() {
int *array, n, i;
srand(time (NULL)); 

printf("Informe o tamanho do array: ");
scanf("%d", &n);

array = (int *) malloc(n * sizeof(int));
if (array == NULL) {
printf("Falha na alocação de memória.\n");
return 1;
}

for(i = 0; i < n; i++)
array[i] = rand () % 100;
for(i = 1; i < n; i++)
printf("\n%d", array[i]);

free(array);
return 0;
}
